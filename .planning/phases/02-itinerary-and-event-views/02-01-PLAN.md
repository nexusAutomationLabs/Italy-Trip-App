1. **event_category enum** (per D-12):
```sql
create type event_category as enum ('dinner', 'excursion', 'group_activity', 'travel', 'open_day');
```

2. **events table**:
```sql
create table public.events (
  id uuid default gen_random_uuid() primary key,
  title text not null,
  description text,
  event_date date not null,
  start_time time,
  end_time time,
  category event_category not null,
  location_name text,
  location_url text,
  created_by uuid references auth.users,
  created_at timestamptz default now()
);
alter table public.events enable row level security;
create policy "Authenticated users can read events" on public.events
  for select to authenticated using (true);
```

3. **rsvps table** (needed for attendee count display per ITIN-04; Phase 3 adds write policies):
```sql
create table public.rsvps (
  id uuid default gen_random_uuid() primary key,
  event_id uuid references public.events on delete cascade not null,
  user_id uuid references auth.users on delete cascade not null,
  created_at timestamptz default now(),
  unique(event_id, user_id)
);
alter table public.rsvps enable row level security;
create policy "Authenticated users can read rsvps" on public.rsvps
  for select to authenticated using (true);
```
